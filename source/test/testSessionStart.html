Here we just define a constructor:

function Person(name, age, g) {
    this.name = name;
	this.age = age || 'unknown';
	this.gender = g || 'unknown'
}
Person.prototype.setSurname = function (s) {
    this.surname = s;
}
Person.prototype.sayHello = function () {
    return "Hello from " + this.name;
}

then we create an instance p

var p = new Person("Federico", 40, 'male');

and after activate encoding (optional), we use instanceBox to store the instance

instanceBox.use('session');
instanceBox.set("Persons/Federico", p);

now to test it, it`s enough to load another page and try to retrieve the instance

(check the "iB-Persons/Federico" value stored in sessionStore)

If everything is correct we should be able to verify the persistence in sessionStorage just loading another page and try to get the data.
