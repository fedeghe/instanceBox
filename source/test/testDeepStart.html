Here we just define a constructor:

function Document(name, age, gender) {
    this.id = [name[0], age, gender, ~~(1E6 * Math.random())].join('')
}

function Person(name, age, g) {
    this.name = name;
    this.age = age || 'unknown';
    this.gender = g || 'unknown';
    this.doc = new Document(name,  age, g)
}
Person.prototype.getDocumentId = function () {
    return this.doc.id;
}

// then we create an instance
var p = new Person("Gabriele", 7, 'male');

finally we use instanceBox to store the instance

instanceBox.use('local');
instanceBox.set("Persons/Gabriele", p);
